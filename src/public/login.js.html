"use strict";
const { Action, api } = require("actionhero");
const bcrypt = require("bcryptjs");
const jwt = require("jsonwebtoken");
const { tokenConfig } = require("../../config/token-config");

/**
 * Web service to login users.
 * @version 1.0.0.
 * @class Login
 * @extends Action
 * @author [Carlos GÃ³mez](https://gitlab.com/charlintosh)
 */
module.exports = class Login extends Action {
  constructor() {
    super();
    this.name = "auth:login";
    this.description = "Provides login credentials (token) for a registered user on KwanModels_Users.";
    this.inputs = {
      email: { required: true },
      password: { required: true }
    };
  }

  async run(data) {

    let params = data.connection.params;
    const User = api.mongooseModels.KwanUsers_User;

    try {
      const user = await User.findOne({ email: params.email });
      if (!user) {
        console.log("Email not found");
        throw new Error("Invalid credentials.");
      }

      // If user is verified, do the normal login
      if (user.verified) {
        const isEqual = await bcrypt.compare(
          params.password,
          user.password
        );
        if (!isEqual) {
          console.log("Incorrect password");
          throw new Error("Invalid credentials.");
        }

        let jwtUser = {
          ...user
        };

        // Token to the server side
        const token = jwt.sign(jwtUser, tokenConfig.secret, {
          expiresIn: tokenConfig.tokenLife
        });

        const refreshToken = jwt.sign(jwtUser, tokenConfig.secret, {
          expiresIn: tokenConfig.tokenLongestLife
        });

        data.response = {
          token: token,
          refreshToken: refreshToken
        };
      } else {
        //Else, tell to the Front end to move at access code page.
        console.log("User registered, need the access");
        data.response = {
          needCode: true
        };
      }
    } catch (e) {
      throw e;
    }
  }
};
